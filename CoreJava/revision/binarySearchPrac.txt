int [] arr = {1,3,4,5,6,7,8,9,10,32,45};
int target = 1;
findIndex(arr,target);

public void findIndex(int [] arr, int target){
  int start = 0;
  int end = arr.length-1;
  while(start<=end){
	int m = start+(end-start)/2;
	if(arr[m]>target){
	  end = m-1;
	}
	else if(arr[m]<target){
	  start = m+1;
	}else{
	    return m;
	}
}
return -1;
}
